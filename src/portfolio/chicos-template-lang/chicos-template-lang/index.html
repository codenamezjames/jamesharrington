Chicos-template-lang

Download this project as a .zip file Download this project as a tar.gz file

Usage

      ${header overwrite="#gps(
        <h1>Header</h1>
        <h1>Header</h1>
      )" }

      <div class="container">
        This is content that should survive
      </div>

      ${footer overwrite="#gps(
        <div>
          <h1>Footer</h1>
        </div>
        <h1>Footer</h1>
      )" }

      <script src="https://jamesharrington.github.io/chicos-template-lang/src/index.js"></script>
      <script>
          document.body.innerHTML = chsParser.render(document.body.innerHTML, {
            header:`
              <div> 
                <div id="gps">
                  <h1>replaceme</h1>
                </div>
              </div>
            `,
            footer:`
              <div> 
                <div id="gps">
                  <h1>replaceme</h1>
                </div>
              </div>
            `
          })
      </script>
